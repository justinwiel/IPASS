79void MPU6050::setup(const MPU6050::sensitiviy &range_setting)
80{
81 uint8_t result;
82 readRegister(WHO_AM_I, &result, 1);
83 ESP_LOGI("MPU", "WHO_AM_I: %X", result);
84 switch (range_setting)
85 {
86 case MPU6050::sensitiviy::high:
87 fs_range = 250;
88 gyrosensitivity = 131; // lsb per second sensitivity values from: https://invensense.tdk.com/wp-content/uploads/2015/02/MPU-6000-Datasheet1.pdf part 6.1 and 6.2
89 accelsensitivity = 16384; // lsb per second
90 break;
91 case MPU6050::sensitiviy::medium:
92 fs_range = 500;
93 gyrosensitivity = 65.5; // lsb per second
94 accelsensitivity = 8192; // lsb per second
95 break;
96 case MPU6050::sensitiviy::low:
97 fs_range = 1000;
98 gyrosensitivity = 32.8; // lsb per second
99 accelsensitivity = 4096; // lsb per second
100 break;
101 case MPU6050::sensitiviy::fullRange:
102 fs_range = 2000;
103 gyrosensitivity = 16.4; // lsb per second
104 accelsensitivity = 2048; // lsb per second
105 break;
106 default:
107 fs_range = 2000;
108 gyrosensitivity = 16.4;
109 accelsensitivity = 2048;
110 }
111 auto to_write = (int(range_setting) << 3); // first three bytes are ignored as such the value needs to be shifted 3 before being written
112 writeRegister(PWR_MGMT_1, 0x80);
113 vTaskDelay(100 / portTICK_PERIOD_MS);
114 writeRegister(PWR_MGMT_1, 0b0001);
115 writeRegister(PWR_MGMT_2, 0x00);
116 vTaskDelay(200 / portTICK_PERIOD_MS);
117 writeRegister(INT_ENABLE, 0x00);
118 writeRegister(FIFO_EN, 0x00);
119 writeRegister(PWR_MGMT_1, 0x00);
120 writeRegister(I2C_MST_CTRL, 0x00);
121 writeRegister(USER_CTRL, 0x00);
122 writeRegister(USER_CTRL, 0x0c);
123 vTaskDelay(15 / portTICK_PERIOD_MS);
124 writeRegister(CONFIG, 0b00000001);
125 writeRegister(SMPLRT_DIV, 0);
126 writeRegister(GYRO_CONFIG, to_write);
127 writeRegister(ACCEL_CONFIG, to_write);
128 writeRegister(USER_CTRL, 0x40);
129 writeRegister(SIGNAL_PATH_RESET, 0b00000111);
130 writeRegister(SIGNAL_PATH_RESET, 0b00000000);
131 writeRegister(GYRO_CONFIG, 0b00011000);
132 writeRegister(ACCEL_CONFIG, 0b00011000);
133 writeRegister(INT_PIN_CFG, 0b00100010);
134}

136void MPU6050::writeRegister(uint8_t sub_addr, uint8_t data)
137{
138 i2c_cmd_handle_t handle = i2c_cmd_link_create();
139 ESP_LOGI("MPU", "Writing to register %X", sub_addr);
140 i2c_master_start(handle);
141 i2c_master_write_byte(handle, (address << 1) | I2C_MASTER_WRITE, true);
142 i2c_master_write_byte(handle, sub_addr, true);
143 i2c_master_write_byte(handle, data, true);
144 i2c_master_stop(handle);
145 i2c_master_cmd_begin(I2C_NUM_0, handle, 1000 / portTICK_PERIOD_MS);
146 i2c_cmd_link_delete(handle);
147}

149uint8_t *MPU6050::readRegister(uint8_t sub_addr, uint8_t *data, uint8_t size)
150{
151
152 i2c_cmd_handle_t handle = i2c_cmd_link_create();
153 i2c_master_start(handle);
154 i2c_master_write_byte(handle, (address << 1) | I2C_MASTER_WRITE, true);
155 i2c_master_write_byte(handle, sub_addr, true);
156 i2c_master_start(handle);
157 i2c_master_write_byte(handle, (address << 1) | I2C_MASTER_READ, true);
158 if (size > 1)
159 {
160 i2c_master_read(handle, data, size - 1, I2C_MASTER_ACK);
161 }
162 i2c_master_read_byte(handle, data + size - 1, I2C_MASTER_NACK);
163 i2c_master_stop(handle);
164 i2c_master_cmd_begin(I2C_NUM_0, handle, 1000 / portTICK_PERIOD_MS);
165 i2c_cmd_link_delete(handle);
166 return data;
167}

169xyz MPU6050::getAccdata_scale(int desired_range)
170{
171 uint8_t data[6];
172 readRegister(ACCEL_XOUT_H, data, 6); // read acclerometer data registers
173 int16_t x = (data[0] << 8) | data[1]; // create 16 bit signed integers from your read 8 bit unsigned integers
174 int16_t y = (data[2] << 8) | data[3];
175 int16_t z = (data[4] << 8) | data[5];
176 return xyz((x / (fs_range / desired_range)) % (desired_range + 1), (y / (fs_range / desired_range)) % (desired_range + 1), (z / (fs_range / desired_range)) % (desired_range + 1));
177}

188void MPU6050::getAccdata_scale(const int &desired_range, xyz *acc)
189{
190 uint8_t data[6];
191 uint16_t range_local = getfs_range();
192 readRegister(ACCEL_XOUT_H, data, 6); // read acclerometer data registers
193 int16_t x = (data[0] << 8) | data[1]; // create 16 bit signed integers from the read 8 bit unsigned integers
194 int16_t y = (data[2] << 8) | data[3];
195 int16_t z = (data[4] << 8) | data[5];
196 acc->x = (x/ (range_local / desired_range)) % (desired_range + 1);
197 acc->y = (y / (range_local / desired_range)) % (desired_range + 1);
198 acc->z = (z / (range_local / desired_range)) % (desired_range + 1);

342all_values MPU6050::fifo_read()
343{
344 fifo_enable(); // open fifo
345 vTaskDelay(50 / portTICK_PERIOD_MS);
346 uint8_t count[2];
347 readRegister(FIFO_COUNTH, count, 2); // read the amount of bytes in fifo buffer
348 int16_t packetcount = (count[0] << 8) | count[1];
349 int16_t data[7];
350 if (packetcount > 0) // don't read if it's empty
351 {
352 uint8_t temp[14];
353 readRegister(FIFO_R_W, temp, 14);
354 data[0] = (temp[0] << 8) | temp[1];
355 data[1] = (temp[2] << 8) | temp[3];
356 data[2] = (temp[4] << 8) | temp[5];
357 data[3] = (temp[6] << 8) | temp[7];
358 data[4] = (temp[8] << 8) | temp[9];
359 data[5] = (temp[10] << 8) | temp[11];
360 data[6] = (temp[12] << 8) | temp[13];
361 }
362 fifo_disable(); // close to prevent overflow
363 return all_values(xyz(data[0] / accelsensitivity, data[1] / accelsensitivity, data[2] / accelsensitivity), xyz(data[4] / gyrosensitivity, data[5] / gyrosensitivity, data[6] / gyrosensitivity), data[3] / 340 + 36.53);
364}
